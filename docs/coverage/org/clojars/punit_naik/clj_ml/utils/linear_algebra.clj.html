006  (defn eval-fn
007    "Evaluates a function ax^n+bx^n-1+...+z represented by a collection of it's coefficients [a b ... z]
008     at the value `x`"
009    [eq x]
010    (reduce + (map-indexed (fn [idx coeff]
011                             (* coeff (Math/pow x (- (dec (count eq)) idx)))) eq)))

013  (defn solve-quadratic-equation
014    "Given the a,b and c terms of the quadratic equation ax^2+bx+c=0
015     This returns a pair of solutions for x"
016    [{:keys [a b c]}]
017    (let [four-a-c (* 4 a c)
018          b-squared (* b b)
019          square-root-b-squared-minus-four-a-c (Math/sqrt (- b-squared four-a-c))
020          two-a (* 2 a)]
021      [(double (/ (+ (* b -1) square-root-b-squared-minus-four-a-c) two-a))
022       (double (/ (+ (* b -1) (* square-root-b-squared-minus-four-a-c -1)) two-a))]))

024  (defn factors
025    "Finds all the factors of a number"
026    ([num]
027     (loop [n (range 1 (inc (Math/abs num)))
028            result #{}]
029       (if (empty? n)
030         result
031         (recur (rest n)
032                (cond-> result
033                  (zero? (mod (Math/abs num) (first n))) (conj (first n)))))))
034    ([num decimal?]
035     (if decimal?
036       (let [[up down] (gu/rationalise (Math/abs num))]
037         (cond-> (factors up)
038           (not= down 1) (intersection (factors down))))
039       (factors num))))

041  (defn isa-solution?
042    "Given the a to z terms of the quadratic equation ax^n+....+z=0 as a collection
043     And the root, this function checks if the same root is a solution for the equation or not
044     And returns the new reduced equation for finding the remaining roots using Synthetic Division"
045    [coefficients root]
046    (let [result (reduce (fn [{:keys [sum] :as acc} v]
047                           (let [s (+ (if (zero? sum) v (* root sum)) (if (zero? sum) sum v))]
048                             (-> acc
049                                 (assoc :sum s)
050                                 (update :coeffs conj s))))
051                         {:sum 0 :coeffs []} coefficients)]
052      (when (and (zero? (Math/round (* (:sum result) 1.0)))
053                 (seq (:coeffs result)))
054        (:coeffs (update result :coeffs butlast)))))

056  (defn find-all-possible-solutions
057    "Given the a to z terms of the equation ax ^n+....+z=0 as a collection
058     This function finds all the possible roots of this equation"
059    [coefficients]
060    (condp = (count coefficients)
061      2 [(/ (* (second coefficients) -1)
062            (first coefficients))]
063      3 (solve-quadratic-equation {:a (nth coefficients 0)
064                                   :b (nth coefficients 1)
065                                   :c (nth coefficients 2)})
066      (let [first-coefficient (first coefficients)
067            last-coefficient (last coefficients)
068            first-coeff-factors (sort (factors first-coefficient true))
069            last-coeff-factors (sort (factors last-coefficient true))]
070        (->> (map (fn [i]
071                    (map (fn [j]
072                           (if (> (Math/abs last-coefficient)
073                                  (Math/abs first-coefficient))
074                             [(/ j i) (/ (* j -1) i)]
075                             [(/ i j) (/ (* i -1) j)]))
076                         last-coeff-factors))
077                  first-coeff-factors)
078             flatten distinct sort))))

080  (defn solve-equation-synthetic-division
081    "Given the a to z terms of the equation ax^n+....+z=0
082     This returns all the roots for the equation using synthetic division"
083    [coefficients]
084    (loop [coeffs coefficients
085           all-possible (reverse (find-all-possible-solutions coeffs))
086           solutions []]
087      (if (or (= (count solutions)
088                 (dec (count coefficients)))
089              (empty? all-possible))
090        (map #(* %1 1.0) solutions)
091        (let [testing-root (first all-possible)
092              next-eq (isa-solution? coeffs testing-root)]
093          (recur (or next-eq coeffs)
094                 (if next-eq
095                   (reverse (find-all-possible-solutions next-eq))
096                   (rest all-possible))
097                 (cond-> solutions
098                   next-eq (conj testing-root)))))))

100  (defn newtons-method
101    "Uses Newton's method to find the root of an equation ax^n+bx^n-1+...+z
102     Represented as a collection of it's coefficients [a b ... z]
103     It selects a root for precision upto the number set by the arg `precision`"
104    [eq eq-deriv precision x-0]
105    (loop [testing-root x-0
106           error 1]
107      (let [f-x (eval-fn eq testing-root)]
108        (if (or (zero? f-x)
109                (<= error (gu/error-decimal precision)))
110        testing-root
111        (let [f-dash-x (eval-fn eq-deriv testing-root)
112              new-root (if (zero? f-dash-x)
113                         ((if (= (/ testing-root (Math/abs testing-root)) -1) + -)
114                          testing-root (gu/error-decimal precision))
115                         (- testing-root (/ f-x f-dash-x)))]
116          (recur new-root (Math/abs (- new-root testing-root))))))))

118  (defn solve-equation-newtons-method
119    "Given the a to z terms of the equation ax^n+....+z=0
120     This returns all the roots for the equation using newton's method"
121    [coefficients]
122    (let [precision 5]
123      (->> (find-all-possible-solutions coefficients)
124           (map #(gu/approximate-decimal
125                  (newtons-method coefficients
126                                  (cu/derivative coefficients)
127                                  precision %)
128                  precision))
129           distinct)))
