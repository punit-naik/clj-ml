002    (:require [org.clojars.punit-naik.clj-ml.utils.generic :as gu]
003              [clojure.set :refer [intersection]]))

005  (defn solve-quadratic-equation
006    "Given the a,b and c terms of the quadratic euqation ax^2+bx+c=0
007     This returns a pair of solutions for x"
008    [{:keys [a b c]}]
009    (let [four-a-c (* 4 a c)
010          b-squared (* b b)
011          square-root-b-squared-minus-four-a-c (Math/sqrt (- b-squared four-a-c))
012          two-a (* 2 a)]
013      [(double (/ (+ (* b -1) square-root-b-squared-minus-four-a-c) two-a))
014       (double (/ (+ (* b -1) (* square-root-b-squared-minus-four-a-c -1)) two-a))]))
015  
016  (defn factors
017    "Finds all the factors of a number"
018    ([num]
019     (loop [n (range 1 (inc (Math/abs num)))
020            result #{}]
021       (if (empty? n)
022         result
023         (recur (rest n)
024                (cond-> result
025                  (zero? (mod (Math/abs num) (first n))) (conj (first n)))))))
026    ([num decimal?]
027     (if decimal?
028       (let [[up down] (gu/rationalise (Math/abs num))]
029         (cond-> (factors up)
030           (not= down 1) (intersection (factors down))))
031       (factors num))))

033  (defn isa-solution?
034    "Given the a to z terms of the quadratic euqation ax^n+....+z=0 as a collection
035     And the root, this function checks if the same root is a solution for thr euqation or not
036     And returns the new reduced equation for finding the remaining roots"
037    [coefficients root]
038    (let [result (reduce (fn [{:keys [sum] :as acc} v]
039                           (let [s (+ (if (zero? sum) v (* root sum)) (if (zero? sum) sum v))]
040                             (-> acc
041                                 (assoc :sum s)
042                                 (update :coeffs conj s))))
043                         {:sum 0 :coeffs []} coefficients)]
044      (when (and (zero? (Math/round (* (:sum result) 1.0)))
045                 (seq (:coeffs result)))
046        (:coeffs (update result :coeffs butlast)))))
047  
048  (defn find-all-possible-solutions
049    "Given the a to z terms of the quadratic euqation ax ^n+....+z=0 as a collection
050     This function finds all the possible roots of this equation"
051    [coefficients]
052    (condp = (count coefficients)
053      2 [(/ (* (second coefficients) -1)
054            (first coefficients))]
055      3 (solve-quadratic-equation {:a (nth coefficients 0)
056                                   :b (nth coefficients 1)
057                                   :c (nth coefficients 2)})
058      (let [first-coefficient (first coefficients)
059            last-coefficient (last coefficients)
060            first-coeff-factors (sort (factors first-coefficient true))
061            last-coeff-factors (sort (factors last-coefficient true))]
062        (->> (map (fn [i]
063                    (map (fn [j]
064                           (if (> (Math/abs last-coefficient)
065                                  (Math/abs first-coefficient))
066                             [(/ j i) (/ (* j -1) i)]
067                             [(/ i j) (/ (* i -1) j)]))
068                         last-coeff-factors))
069                  first-coeff-factors)
070             flatten distinct sort))))
071  
072  (defn solve-equation
073    "Given the a to z terms of the quadratic euqation ax^n+....+z=0
074     This returns all the roots for the equation"
075    [coefficients]
076    (loop [coeffs coefficients
077           all-possible (reverse (find-all-possible-solutions coeffs))
078           solutions []]
079      (if (or (= (count solutions)
080                 (dec (count coefficients)))
081              (empty? all-possible))
082        (map #(* %1 1.0) solutions)
083        (let [testing-root (first all-possible)
084              next-eq (isa-solution? coeffs testing-root)]
085          (recur (or next-eq coeffs)
086                 (if next-eq
087                   (reverse (find-all-possible-solutions next-eq))
088                   (rest all-possible))
089                 (cond-> solutions
090                   next-eq (conj testing-root)))))))
