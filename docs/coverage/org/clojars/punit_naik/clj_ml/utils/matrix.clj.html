252    "Converts any square matrix into an upper-triangular matrix
253     where all the matrix elements below the diagonal elements are zero"
254    [matrix]
255    (let [[num-rows _] (dimension matrix)]
256      (loop [m matrix
257             nr (range 1 num-rows)
258             swap-count 0]
259        (if (empty? nr)
260          {:upper-triangular m :num-swaps swap-count}
261          (let [adjusted-row (recursive-row-adjust m (first nr))
262                first-n-zeros-adjusted-row (gu/first-n-zeros adjusted-row)
263                m-adjusted (gu/replace-nth m (first nr) adjusted-row)
264                should-be-swapped? (> first-n-zeros-adjusted-row (first nr))
265                m-swapped (cond-> m-adjusted
266                            should-be-swapped? (swap-rows (first nr) first-n-zeros-adjusted-row))]
267            (recur m-swapped
268                   (cond-> nr
269                     (not should-be-swapped?) rest)
270                   (cond-> swap-count
271                     should-be-swapped? inc)))))))
