001  (ns org.clojars.punit-naik.clj-ml.utils.string)
002  
003  (defn levenshtein-distance
004    "Finds the percentage of match (reversed levenshtein distance & normalised)
005     between two strings"
006    [s1 s2]
007    (let [similarity (atom 0)
008          edit-map (atom {})]
009      (doseq [i (range (count s1))]
010        (let [edit (atom (if (zero? i) 0 i))]
011          (doseq [j (range (count s2))]
012            (let [e (if (= (.charAt s1 i) (.charAt s2 j)) 0 1)]
013              (reset! edit
014                      (min
015                       (if (zero? j)
016                         (+ i 2)
017                         (inc @edit))
018                       (if (zero? i)
019                         (+ j 2)
020                         (inc (@edit-map (str (dec i) "," j))))
021                       (+ (cond
022                            (zero? i) j
023                            (zero? j) i
024                            :else (@edit-map (str (dec i) "," (dec j))))
025                          (if (and (zero? i) (zero? j)) 0 e))))
026              (swap! edit-map assoc (str i "," j) @edit)
027              (when (and (= i (dec (count s1))) (= j (dec (count s2))))
028                (reset! edit-map nil)
029                (reset! similarity @edit))))))
030      (double (- 100 (* 100 (/ @similarity (+ (count s1) (count s2))))))))
